# Scene Dimensions
SCENE_SIZE = 100.0
Z_OFFSET = 20.0

# Landscape Settings
subdivisions = 800
factor = 5.0
water_level = -0.5
frequency = 0.25
octaves = 4
octave_effect = 1
height = 0.0
TREE_MAP = {}

# Water Animation Settings
water_noise_effect = 10
water_noise_frequency = 10
water_wave_frequency = 400
wave_height = 1/200

# Lighting & Animation Settings
max_sun_energy = 3
pyramid_animation_frames = 200

# Tree Settings
num_trees = 100
tree_radius = .5
# Pyramid Settings
subdivision_depth = 2 # Recursive, More = Longer Generation

# Scene Cleanup
def delete_all_objects():
    """Cleans the scene by deleting all Meshes, Lights, and Empties."""
    if bpy.context.active_object and bpy.context.active_object.mode != 'OBJECT':
        bpy.ops.object.mode_set(mode='OBJECT')

    bpy.ops.object.select_all(action='DESELECT')
    bpy.ops.object.select_by_type(type='MESH')
    bpy.ops.object.delete()

    bpy.ops.object.select_all(action='DESELECT')
    for obj in bpy.context.scene.objects:
        if obj.type in {'LIGHT', 'EMPTY'}:
            obj.select_set(True)
    
    if any(obj.select_get() for obj in bpy.context.scene.objects):
        bpy.ops.object.delete()
    
    bpy.ops.object.select_all(action='DESELECT')

def make_vertex_color_material():
    """Creates a material node setup to display vertex colors."""
    mat = bpy.data.materials.new(name="VertexColorMaterial")
    mat.use_nodes = True
    bsdf = mat.node_tree.nodes.get("Principled BSDF")
    color_node = mat.node_tree.nodes.new(type='ShaderNodeVertexColor')
    color_node.layer_name = "Col"
    mat.node_tree.links.new(color_node.outputs['Color'], bsdf.inputs['Base Color'])
    return mat

def make_key(co):
    """
    Snaps a random (x, y) coordinate to the nearest available grid vertex.
    This is critical for successful lookup in the TREE_MAP.
    """
    step = SCENE_SIZE / subdivisions
    x = round(co[0] / step) * step
    y = round(co[1] / step) * step
    return (round(x, 3), round(y, 3))

def create_textured_material(mat_name, image_filename):
    """Creates a material and attempts to load an image texture."""
    material = bpy.data.materials.new(name=mat_name)
    material.use_nodes = True
    bsdf_node = material.node_tree.nodes.get("Principled BSDF")
    texture_node = material.node_tree.nodes.new(type="ShaderNodeTexImage")
    
    try:
        texture_node.image = bpy.data.images.load(f"//{image_filename}")
        material.node_tree.links.new(texture_node.outputs["Color"], bsdf_node.inputs["Base Color"])
    except RuntimeError:
        print(f"Warning: Image '{image_filename}' not found. Material '{mat_name}' will be blank.")
        
    return material

# Water Grid Generation
bpy.ops.mesh.primitive_grid_add(size=SCENE_SIZE, x_subdivisions=subdivisions, y_subdivisions=subdivisions, location=(0, 0, water_level))
obj_water = bpy.context.active_object
for v in obj_water.data.vertices:
    v.co.z += sin(noise.noise(v.co * water_noise_frequency)*water_noise_effect + v.co.x * water_wave_frequency) * wave_height
obj_water.data.materials.append(make_vertex_color_material())

# Apply colors to water (Murky Blue)
bpy.ops.object.mode_set(mode='EDIT')
bm = bmesh.from_edit_mesh(obj_water.data)
bm.faces.ensure_lookup_table()
color_layer = bm.loops.layers.color.new("Col")
for face in bm.faces:
    for loop in face.loops:
        loop[color_layer] = (0.05, 0.05, 0.7, .3)
bmesh.update_edit_mesh(obj_water.data)
bpy.ops.object.mode_set(mode='OBJECT')

# Terrain Grid Generation
bpy.ops.mesh.primitive_grid_add(size=SCENE_SIZE, x_subdivisions=subdivisions, y_subdivisions=subdivisions)
obj_terrain = bpy.context.active_object

# Generate noise terrain height and populate tree_map
for v in obj_terrain.data.vertices:
    v.co.z += height + (noise.noise(frequency * v.co) * factor)
    for o in range(octaves):
        v.co.z += noise.noise((2**(o+1))*frequency * v.co) * octave_effect * factor/(2**(o+2))
    
    TREE_MAP[make_key((v.co.x, v.co.y))] = v.co.z

obj_terrain.data.materials.append(make_vertex_color_material())

# Apply colors to terrain
bpy.ops.object.mode_set(mode='EDIT')
bm = bmesh.from_edit_mesh(obj_terrain.data)
bm.faces.ensure_lookup_table()
color_layer = bm.loops.layers.color.new("Col")
sand = (0.35, 0.3, 0.15, 1)    # Muted, muddy brown/sand
dark_rock = (0.1, 0.1, 0.05, 1) # Dark, peat/algae-covered rock

for face in bm.faces:
    for loop in face.loops:
        elev = loop.vert.co.z
        
        # Color based on elevation
        if elev < water_level:
            loop[color_layer] = (0.15, 0.1, 0.05, 1) 
        elif elev < water_level + 0.5:
            loop[color_layer] = sand
        elif elev < factor * 0.5:
            loop[color_layer] = sand
        else:
            t = (elev - factor * 0.5) / (factor * 0.5)
            r = sand[0] * (1-t) + dark_rock[0] * t
            g = sand[1] * (1-t) + dark_rock[1] * t
            b = sand[2] * (1-t) + dark_rock[2] * t
            loop[color_layer] = (r, g, b, 1)

bmesh.update_edit_mesh(obj_terrain.data)
bpy.ops.object.mode_set(mode='OBJECT')

# --- FRACTAL PYRAMID GENERATION ---
def create_cylinder_for_edge(v1, v2, radius):
    """Creates an individual cylinder segment for the fractal structure."""
    vec = v2 - v1
    length = vec.length
    midpoint = (v1 + v2) / 2

    bpy.ops.mesh.primitive_cylinder_add(
        vertices=4, radius=radius, depth=length, location=midpoint
    )
    cylinder = bpy.context.object
    quat = vec.normalized().to_track_quat('Z', 'Y')
    cylinder.rotation_mode = 'QUATERNION'
    cylinder.rotation_quaternion = quat
    return cylinder

def subdivide_pyramid(pyramid_points, depth_remaining, radius, all_cylinders):
    """Recursive function to generate the fractal based on subdivision_depth."""
    tip, c1, c2, c3, c4 = [mathutils.Vector(p) for p in pyramid_points]

    if depth_remaining < 1:
        # Draw the final edges as cylinders
        edges = [ (tip, c1), (tip, c2), (tip, c3), (tip, c4), (c1, c2), (c2, c3), (c3, c4), (c4, c1) ]
        for v1, v2 in edges:
            all_cylinders.append(create_cylinder_for_edge(v1, v2, radius))
        return

    # Calculate midpoints for 5 new sub-pyramids
    mid_c1_c2 = (c1 + c2) / 2; mid_c2_c3 = (c2 + c3) / 2; mid_c3_c4 = (c3 + c4) / 2; mid_c4_c1 = (c4 + c1) / 2
    mid_tip_c1 = (tip + c1) / 2; mid_tip_c2 = (tip + c2) / 2; mid_tip_c3 = (tip + c3) / 2; mid_tip_c4 = (tip + c4) / 2
    center_base = (c1 + c2 + c3 + c4) / 4
    
    pyramids_to_create = [
        [tip, mid_tip_c1, mid_tip_c2, mid_tip_c3, mid_tip_c4],
        [mid_tip_c1, c1, mid_c1_c2, center_base, mid_c4_c1],
        [mid_tip_c2, c2, mid_c2_c3, center_base, mid_c1_c2],
        [mid_tip_c3, c3, mid_c3_c4, center_base, mid_c2_c3],
        [mid_tip_c4, c4, mid_c4_c1, center_base, mid_c3_c4]
    ]
    
    for new_pyramid_points in pyramids_to_create:
        subdivide_pyramid(new_pyramid_points, depth_remaining - 1, radius=0.35, all_cylinders=all_cylinders)

def color_fractal(obj, num_colors):
    """Creates and assigns random materials to the fractal mesh polygons."""
    if obj.type != 'MESH': return
    obj.data.materials.clear()
    
    for i in range(num_colors):
        mat = bpy.data.materials.new(name=f"RandomColor_{i}")
        mat.use_nodes = True
        principled_bsdf = mat.node_tree.nodes.get('Principled BSDF')
        if principled_bsdf:
            principled_bsdf.inputs['Base Color'].default_value = (random.random(), random.random(), random.random(), 1.0)
            principled_bsdf.inputs['Emission Strength'].default_value = 0.0 
        obj.data.materials.append(mat)
    
    for poly in obj.data.polygons:
        poly.material_index = random.randint(0, num_colors - 1)

def generate_pyramid_fractal():
    """Generates the base pyramid, recursively subdivides it, joins all segments, and colors it."""
    size_mult = 25
    initial_pyramid_points = [
        (0 * size_mult, 0 * size_mult, 1 * size_mult + Z_OFFSET),
        (-1 * size_mult, -1 * size_mult, 0 * size_mult + Z_OFFSET),
        (1 * size_mult, -1 * size_mult, 0 * size_mult + Z_OFFSET),
        (1 * size_mult, 1 * size_mult, 0 * size_mult + Z_OFFSET),
        (-1 * size_mult, 1 * size_mult, 0 * size_mult + Z_OFFSET)
    ]

    all_cylinders = []
    subdivide_pyramid(initial_pyramid_points, subdivision_depth, radius=0.35, all_cylinders=all_cylinders)

    if all_cylinders:
        # Join into a single object
        bpy.context.view_layer.objects.active = all_cylinders[0]
        for obj in all_cylinders: obj.select_set(True)
        bpy.context.view_layer.objects.active = all_cylinders[0]
        bpy.ops.object.join()
        joined_object = bpy.context.view_layer.objects.active
        
        color_fractal(joined_object, 50) 
        joined_object.select_set(False)
        return joined_object
    return None

# --- ANIMATION & LIGHTING SETUP ---
def setup_lights_and_animations(fractal_object):
    """Sets the frame range, animates the sun (rise/set), and animates the pyramid (spin)."""
    
    start_frame = 1
    end_frame = pyramid_animation_frames
    bpy.context.scene.frame_start = start_frame
    bpy.context.scene.frame_end = end_frame
    
    # === SUN LIGHT ANIMATION (Sunrise/Sunset Driver) ===
    bpy.ops.object.empty_add(location=(0, 0, 0))
    light_rotator_sun = bpy.context.active_object
    light_rotator_sun.name = "Light_Rotator_Sun"

    light_data = bpy.data.lights.new(name="SunLightData", type='SUN')
    light_data.energy = max_sun_energy
    light_data.color = (1.0, 1.0, 0.9)  

    sun = bpy.data.objects.new(name="AnimatedSun", object_data=light_data)
    bpy.context.collection.objects.link(sun)
    
    sun_distance = 100.0
    sun.location = (sun_distance, 0, 0)
    sun.parent = light_rotator_sun
    sun.rotation_euler[1] = radians(90) # Point sun towards center

    # Driver rotates the empty based on frame number
    driver = light_rotator_sun.driver_add('rotation_euler', 1).driver 
    driver.type = 'SCRIPTED'
    driver.expression = '-(frame * (pi / 200))'
    
    # === PYRAMID ROTATION (Spin) ===
    if fractal_object:
        # Empty for the pyramid to rotate around its base origin
        bpy.ops.object.empty_add(location=(0, 0, Z_OFFSET)) 
        pyramid_rotator = bpy.context.active_object
        pyramid_rotator.name = "Pyramid_Rotator"
        fractal_object.parent = pyramid_rotator

        # Keyframe Rotation from 0 to 2*pi (360 degrees) on the Z-axis
        pyramid_rotator.rotation_euler = (0, 0, 0)
        pyramid_rotator.keyframe_insert(data_path="rotation_euler", frame=start_frame)
        
        pyramid_rotator.rotation_euler = (0, 0, 2 * math.pi) 
        pyramid_rotator.keyframe_insert(data_path="rotation_euler", frame=end_frame)
        
        # Linear for constant spin speed
        if pyramid_rotator.animation_data and pyramid_rotator.animation_data.action:
             for fcurve in pyramid_rotator.animation_data.action.fcurves:
                 if fcurve.data_path == "rotation_euler":
                     for keyframe in fcurve.keyframe_points:
                         keyframe.interpolation = 'LINEAR'

    bpy.context.scene.frame_set(start_frame)


# --- TREE GENERATION ---
def new_tree(location):
    """Generates a single fractal tree mesh at the given location."""
    verts = []
    faces = []
    face_uvs = []
    cylinder_sections = 30

    def make_end_ring(start, end, radius, start_instead=False):
        # Creates a ring of vertices perpendicular to the segment (start to end)
        middle_vector = end - start
        point_one = middle_vector.cross((1, 0, 0)).normalized() * radius
        point_two = middle_vector.cross(point_one).normalized() * radius
        
        ring_indices = []
        for s in range(cylinder_sections + 1):
            angle_a = s * 2 * pi / cylinder_sections
            edge_point = cos(angle_a) * point_one + sin(angle_a) * point_two + start
            
            ring_indices.append(len(verts))
            if start_instead:
                verts.append(edge_point)
            else:
                verts.append(edge_point + middle_vector)
        return ring_indices
    
    def make_section(start_ring, end_ring):
        # Creates the faces between two rings (a cylindrical segment)
        for i in range(cylinder_sections):
            faces.append([start_ring[i], end_ring[i], end_ring[i+1], start_ring[i+1]])

    def add_texture_coordinates(tree_mesh):
        # Sets basic UV coordinates for bark texture wrapping
        uv_layer = tree_mesh.uv_layers.new(name="Bark")
        # UV logic would be complex here, simplified for display purposes
        
    def draw_branch(start, end, radius, prev_pos, subdivisions, start_ring):
        # Draws a curved segment using 4 cylinder sections
        section_length = (start - prev_pos).length
        continue_vector = section_length * (start - prev_pos).normalized()

        middle_of_straight_branch = (start + end) / 2
        end_of_continue = start + continue_vector
        half_continue = start + continue_vector/2
        
        # Interpolate points for the curve
        early_spot = (2 * start + middle_of_straight_branch + 2 * half_continue ) / 5
        middle_spot = (2 * start + end + end_of_continue) / 4
        late_spot = (middle_spot + 2 * end + middle_of_straight_branch) / 4
        
        # Draw and connect the segments
        r_mult = 1.0
        early_ring = make_end_ring(start, early_spot, radius * 1.2 * r_mult)
        middle_ring = make_end_ring(early_spot, middle_spot, radius * 1.1 * r_mult)
        late_ring = make_end_ring(middle_spot, late_spot, radius * 1.05 * r_mult)
        end_ring = make_end_ring(late_spot, end, radius * r_mult)
        
        make_section(start_ring, early_ring)
        make_section(early_ring, middle_ring)
        make_section(middle_ring, late_ring)
        make_section(late_ring, end_ring)
        return end_ring
    
    def make_tree(pos, calls, prev_pos, prev_ring, stop_adjustment=1):
        # Recursive function for branching
        if random.random() > stop_adjustment / calls:
            return

        branches = random.randint(1, 3) 
        for i in range(branches):
            # Calculate a new endpoint for the branch
            new_endpoint = Vector((
                pos.x + 2 * (random.random() - 0.5), 
                pos.y + 2 * (random.random() - 0.5), 
                pos.z + 1 + random.random()
            ))
            new_end_ring = draw_branch(pos, new_endpoint, tree_radius/calls, prev_pos, 1, prev_ring)
            make_tree(new_endpoint, calls + 1, pos, new_end_ring, stop_adjustment=stop_adjustment)
    
    # --- Tree Construction ---
    # Create the initial ring and trigger recursive branching
    start_ring = make_end_ring(Vector((0, 0, 0)), Vector((0, 0, 2)), tree_radius, start_instead=True)
    make_tree(Vector((0, 0, 0)), 1, Vector((0, 0, -2)), start_ring, 4)

    # Final mesh object
    tree_mesh = bpy.data.meshes.new(name="TreeMesh")
    tree_mesh.from_pydata(vertices=verts, edges=[], faces=faces)
    add_texture_coordinates(tree_mesh)
    tree_mesh.validate(verbose=False)  
    
    tree_object = bpy.data.objects.new(name="Tree", object_data=tree_mesh)
    bpy.context.collection.objects.link(tree_object)

    # Move the tree to the desired spot on the terrain
    tree_object.location = location 
    
    # Apply Bark Material
    mat = create_textured_material("TreeBark", "bark.jpg")
    tree_object.data.materials.append(mat)
    
    return tree_object

def generate_forest():
    """Iterates to place NUM_TREES randomly across the landscape, avoiding water."""
    for i in range(num_trees):
        # Select a random coordinate (x, y) within the scene bounds
        x = random.uniform(-SCENE_SIZE / 2, SCENE_SIZE / 2)
        y = random.uniform(-SCENE_SIZE / 2, SCENE_SIZE / 2)
        
        # Get the snapped key and elevation from the map
        key = make_key((x, y))
        
        # Placement Check: Only create the tree if the ground is above the water level
        if key in TREE_MAP and TREE_MAP[key] > water_level:
            z = TREE_MAP[key]
            
            # Create the tree at the ground location (x, y, z)
            new_tree(location=(x, y, z))
